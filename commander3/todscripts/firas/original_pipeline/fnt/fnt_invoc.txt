	character * 64 ftc_plt_com_file	!PLT command file name
	integer * 4 ftc_plots
	integer * 4 ftc_jump

	character 	*14	ftc_jstart_time

	character 	*14	ftc_jstop_time
	integer 	*4 	ftc_jstop(2)
	character 	*30  	ftc_time_range

	common /ftc/	ftc_plot, ftc_preamp, ftc_file_name, ftc_write,

     .			ftc_jump, ftc_display, ftc_interactive, ftc_input,
